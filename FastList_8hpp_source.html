188 FastList(size_t size)
189 {
190 int node_idx = 0;
191 nodes_.reserve(size);
192 for(size_t i = 0; i < size; ++i) {
193 Node n(node_idx);
194 n.prev = node_idx - 1;
195 n.next = node_idx + 1;
196 ++node_idx;
197 nodes_.emplace_back(n);
198 }
199 nodes_.back().next = -1;

245 iterator erase(const const_iterator & entry)
246 {
247 const auto node_idx = entry.getIndex();
248 auto & node_to_erase = nodes_[node_idx];
249 reinterpret_cast<T*>(&node_to_erase.type_storage)->~T();
250 int next_elem = -1;
251
252 if(first_node_ == node_idx) {
253 first_node_ = node_to_erase.next;
254 }
255 if(last_node_ == node_idx) {
256 last_node_ = node_to_erase.prev;
257 }
258
259 if(SPARTA_EXPECT_FALSE(node_to_erase.next != -1))
260 {
261 auto & next_node = nodes_[node_to_erase.next];
262 next_node.prev = node_to_erase.prev;
263 next_elem = node_to_erase.next;
264 }
265
266 if(SPARTA_EXPECT_FALSE(node_to_erase.prev != -1))
267 {
268 auto & prev_node = nodes_[node_to_erase.prev];
269 prev_node.next = node_to_erase.next;
270 }
271
272 node_to_erase.prev = -1;
273 node_to_erase.next = -1;
274 if(SPARTA_EXPECT_TRUE(free_head_ != -1)) {
275 nodes_[free_head_].prev = node_idx;
276 node_to_erase.next = free_head_;
277 }
278 free_head_ = node_idx;
279 --size_;
280 return iterator(this, next_elem);
281 }

283 template<class ...ArgsT>
284 iterator emplace(const const_iterator & pos, ArgsT&&...args)
285 {
286 sparta_assert(free_head_ != -1,
287 "FastList is out of element room");
288 const auto index_pos = pos.getIndex();
289
290 // If the index pos is -1, it's either end() or begin() on
291 // an empty list. Just emplace_back (or front, don't matter)
292 if(index_pos == -1) {
293 return emplace_back(std::forward<ArgsT>(args)...);
294 }
295
296 auto & new_node = nodes_[free_head_];
297 free_head_ = new_node.next;
298 new (&new_node.type_storage) T(args...);
299 // Update pointers. Start with a clean slate
300 new_node.next = -1;
301 new_node.prev = -1;
302
303 // Insert before the given pt
304 auto & insert_pt = nodes_[index_pos];
305 new_node.next = insert_pt.index;
306 new_node.prev = insert_pt.prev;
307 insert_pt.prev = new_node.index;
308 if(new_node.prev != -1) {
309 // update the previous node's next
310 nodes_[new_node.prev].next = new_node.index;
311 }
312
313 if((first_node_ == index_pos) || (first_node_ == -1))
314 {
315 first_node_ = new_node.index;
316 }
317 ++size_;
318
319 return iterator(this, new_node.index);
320 }
